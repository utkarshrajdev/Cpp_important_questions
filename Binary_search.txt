#include<bits/stdc++.h>
using namespace std;

int find(int arr[], int e, int p){
	int s=0;
	int mid=0;
	while(s<e)
	{
		mid=(e-s)/2+s;
		if(arr[mid]==p) return mid;
		else if(arr[mid]<p) s=mid+1;
		else e=mid-1;
	}
	return -1;
}
int main(){
    int n,p;
    cout<<"Enter size: ";
    cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
	{
	    cout<<"Enter value for "<< i<<"th element: ";
	    cin>>arr[i];
	}
	cout<<"Enter number to find: ";
	cin>>p;
	int ans= find(arr,n,p);
	cout<<"Output: "<<ans;
	return 0;
}
